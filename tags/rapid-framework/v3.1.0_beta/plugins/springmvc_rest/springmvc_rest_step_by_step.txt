		UrlRewriteFilter rule
			<rule>
		    	<condition operator="notequal" next="and" type="request-uri">.*.jsp</condition>
		    	<condition operator="notequal" next="and" type="request-uri">.*.jspx</condition>
		        <from>^(/.*\..*)$</from>
		        <to>/static$1</to>
		    </rule>
