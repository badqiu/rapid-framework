	private boolean isFoundInsertLocation(Template template, Map model, File outputFile, StringWriter newFileContentCollector) throws IOException, TemplateException {
		LineNumberReader reader = new LineNumberReader(new FileReader(outputFile));
		PrintWriter writer = new PrintWriter(newFileContentCollector);
		
		try {
			String fileContent =  IOHelper.readFile(outputFile);
			int indexOfInsertLocation = fileContent.indexOf(GENERATOR_INSERT_LOCATION);
			if(indexOfInsertLocation >= 0) {
				ByteArrayOutputStream templateOutput = new ByteArrayOutputStream();
				template.process(model, new OutputStreamWriter(templateOutput,encoding));
				if(fileContent.contains(templateOutput.toString())) {
					//no insert
					writer.append(fileContent);
				}else {
					//do insert
					StringBuffer insertedOutput = new StringBuffer(fileContent).insert(indexOfInsertLocation,templateOutput);
					writer.append(insertedOutput.toString());
				}
				return true;
			}else {
				return false;
			}
		}finally {
			writer.close();
			reader.close();			
		}
	}
